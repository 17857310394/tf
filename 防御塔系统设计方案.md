# 防御塔系统设计方案

## 1. 防御塔类型系统设计

### 1.1 基础防御塔类型定义

#### 1.1.1 普通塔（GroundAttackTower）
- **特性**：基础物理攻击，单体伤害
- **攻击方式**：发射子弹，直线飞行
- **攻击目标**：地面敌人
- **特殊效果**：无
- **建造成本**：100金币
- **升级成本**：每级增加50金币
- **视觉特征**：简洁的炮塔设计，发射普通子弹

#### 1.1.2 导弹塔（MissileTower）
- **特性**：高伤害，追踪能力
- **攻击方式**：发射导弹，具有追踪效果
- **攻击目标**：地面和空中敌人
- **特殊效果**：导弹命中后产生小范围爆炸
- **建造成本**：200金币
- **升级成本**：每级增加100金币
- **视觉特征**：较大的导弹发射器，发射追踪导弹

#### 1.1.3 激光塔（LaserTower）
- **特性**：持续伤害，穿透能力
- **攻击方式**：发射激光，持续照射目标
- **攻击目标**：地面敌人
- **特殊效果**：激光可以穿透多个敌人
- **建造成本**：150金币
- **升级成本**：每级增加75金币
- **视觉特征**： futuristic设计，发射激光束

#### 1.2.1 塔基类设计
```gdscript
class_name BaseTower extends Node3D

# 核心属性
var tower_data: TowerData
var level: int = 1
var current_health: float
var attack_cooldown: float = 0

# 状态
var is_attacking: bool = false
var target: Node3D = null

# 信号
signal on_target_acquired(target)
signal on_target_lost()
signal on_attack_started()
signal on_attack_completed()
signal on_upgraded(new_level)
signal on_sold()
signal on_destroyed()

# 初始化
func _ready():
    current_health = tower_data.base_health

# 更新
func _process(delta):
    attack_cooldown = max(0, attack_cooldown - delta)
    if attack_cooldown <= 0 and target:
        attack()

# 攻击方法（由子类实现）
func attack():
    pass

# 目标选择（由子类实现）
func select_target():
    pass

# 升级方法
func upgrade():
    if level < tower_data.max_level:
        level += 1
        current_health = get_max_health()
        emit_signal("on_upgraded", level)

# 获取最大生命值
func get_max_health() -> float:
    return tower_data.base_health * pow(tower_data.health_growth_rate, level - 1)

# 获取当前攻击力
func get_current_damage() -> float:
    return tower_data.base_damage * pow(tower_data.damage_growth_rate, level - 1)

# 获取当前攻击速度
func get_current_attack_speed() -> float:
    return tower_data.base_attack_speed / pow(tower_data.attack_speed_growth_rate, level - 1)

# 获取当前攻击范围
func get_current_range() -> float:
    return tower_data.base_range * pow(tower_data.range_growth_rate, level - 1)

# 承受伤害
func take_damage(amount: float) -> void:
    current_health = max(0, current_health - amount)
    if current_health <= 0:
        emit_signal("on_destroyed")
        queue_free()

# 出售
func sell() -> int:
    var sell_value = int(tower_data.build_cost * 0.7 * pow(1.2, level - 1))
    emit_signal("on_sold")
    queue_free()
    return sell_value
```

#### 1.2.2 防御塔注册机制
```gdscript
# TowerRegistry.gd (Autoload)
class_name TowerRegistry

var tower_types = {}

# 注册防御塔类型
func register_tower_type(type_name: String, tower_script: Script, tower_data: TowerData) -> void:
    tower_types[type_name] = {
        "script": tower_script,
        "data": tower_data
    }

# 获取防御塔类型
func get_tower_type(type_name: String) -> Dictionary:
    return tower_types.get(type_name, null)

# 获取所有防御塔类型
func get_all_tower_types() -> Array:
    return tower_types.keys()

# 创建防御塔实例
func create_tower(type_name: String, position: Vector3) -> BaseTower:
    var tower_info = get_tower_type(type_name)
    if not tower_info:
        return null
    
    var tower = tower_info.script.new()
    tower.tower_data = tower_info.data
    tower.position = position
    return tower
```

### 1.3 攻击类型分类体系

#### 1.3.1 地面攻击型（GroundAttackType）

- **攻击目标**：地面和空中敌人
- **代表防御塔**：导弹塔
- **实现方式**：检测范围包含空中敌人

#### 1.3.3 攻击目标判定
```gdscript
# 在BaseTower中添加
func can_attack_target(target: Node3D) -> bool:
    if not target:
        return false
    
    var target_type = target.get("enemy_type", "ground")
    match tower_data.attack_type:
        "ground":
            return target_type == "ground"
        "air":
            return target_type == "air"
        "both":
            return true
    return false
```

```gdscript
# TowerData.gd (Resource)
class_name TowerData extends Resource

# 基础属性
export var name: String = "Tower"
export var description: String = "A basic tower"
export var build_cost: int = 100
export var base_damage: float = 10.0
export var base_health: float = 100.0
export var base_attack_speed: float = 1.0  # 攻击间隔（秒）
export var base_range: float = 10.0
export var attack_type: String = "ground"  # ground, air, both

# 升级属性
export var max_level: int = 3
export var damage_growth_rate: float = 1.5
export var health_growth_rate: float = 1.3
export var attack_speed_growth_rate: float = 1.2
export var range_growth_rate: float = 1.1
export var upgrade_cost_base: int = 50
export var upgrade_cost_multiplier: float = 1.5

# 视觉属性
export var scene_path: String = ""
export var icon_path: String = ""

# 特殊技能
export var skills: Array[Dictionary] = []

# 获取升级成本
func get_upgrade_cost(level: int) -> int:
    if level >= max_level:
        return 0
    return int(upgrade_cost_base * pow(upgrade_cost_multiplier, level - 1))

# 验证数据
func validate() -> bool:
    if name.is_empty():
        return false
    if build_cost <= 0:
        return false
    if base_damage <= 0:
        return false
    if base_health <= 0:
        return false
    if base_attack_speed <= 0:
        return false
    if base_range <= 0:
        return false
    if max_level < 1:
        return false
    return true
```

```gdscript
var skills = {}

            "data": skill_data,

            # 实现范围伤害
            pass
        "slow":
            # 实现减速效果
            pass
        "stun":
            # 实现眩晕效果
            pass
        "poison":
            # 实现中毒效果
            pass
    
    # 设置冷却时间
```

## 3. 需要进一步明确的内容

### 3.1 防御塔的目标选择逻辑与优先级规则

#### 3.1.1 目标选择逻辑
```gdscript
# 在BaseTower中实现
func select_target():
    var area = $Area3D
    var candidates = []
    
    # 获取范围内的敌人
    for body in area.get_overlapping_bodies():
        if body.is_in_group("enemies") and can_attack_target(body):
            candidates.append(body)
    
    if candidates.empty():
        target = null
        emit_signal("on_target_lost")
        return
    
    # 按照优先级排序
    candidates.sort_custom(self, "_compare_targets")
    target = candidates[0]
    emit_signal("on_target_acquired", target)

func _compare_targets(a: Node3D, b: Node3D) -> int:
    # 优先级1：距离最近（默认）
    var distance_a = global_position.distance_to(a.global_position)
    var distance_b = global_position.distance_to(b.global_position)
    if distance_a < distance_b:
        return -1
    elif distance_a > distance_b:
        return 1
    
    # 优先级2：生命值最低（保留逻辑）
    var health_a = a.get("health", 0)
    var health_b = b.get("health", 0)
    if health_a < health_b:
        return -1
    elif health_a > health_b:
        return 1
    
    # 优先级3：移动速度最快（保留逻辑）
    var speed_a = a.get("speed", 0)
    var speed_b = b.get("speed", 0)
    if speed_a > speed_b:
        return -1
    elif speed_a < speed_b:
        return 1
    
    return 0
```

```gdscript
var range_visualizer: MeshInstance3D

func create_range_visualizer():
    range_visualizer = MeshInstance3D.new()
```

### 3.4 特殊技能的具体分类与实现方式

#### 3.4.1 技能分类
- **主动技能**：需要玩家手动触发的技能
- **被动技能**：自动触发的技能
- **持续技能**：持续一段时间的技能效果
- **瞬时技能**：立即产生效果的技能

#### 3.4.2 技能效果类型
- **范围伤害**：对范围内的敌人造成伤害
- **减速**：降低敌人的移动速度
- **眩晕**：使敌人暂时无法移动和攻击
- **中毒**：持续对敌人造成伤害
- ** buff**：提升防御塔的属性
- ** debuff**：降低敌人的属性

#### 3.4.3 技能实现示例
```gdscript
# 范围伤害技能
func trigger_area_damage(skill_data: Dictionary):
    var area = Area3D.new()
    var shape = SphereShape3D.new()
    shape.radius = skill_data.range
    area.add_shape(shape)
    area.global_position = global_position
```
